    <script>
        const API_URL = 'http://localhost:3000/api';
        let token = localStorage.getItem('token');
        let adWatching = false;

        if (!token) {
            window.location.href = '/';
        }

        async function loadProfile() {
            try {
                const response = await fetch(`${API_URL}/user/profile`, {
                    headers: { 'Authorization': `Bearer ${token}` }
                });
                const data = await response.json();

                if (data.success) {
                    const user = data.user;
                    document.getElementById('username').textContent = user.username;
                    document.getElementById('balance').textContent = user.balance.toFixed(2);
                    document.getElementById('withdrawBalance').textContent = user.balance.toFixed(2);
                    document.getElementById('totalEarned').textContent = user.total_earned.toFixed(2);
                    document.getElementById('adsToday').textContent = user.adsWatchedToday;
                    document.getElementById('adsWatched').textContent = user.adsWatchedToday;
                    document.getElementById('adsRemaining').textContent = 20 - user.adsWatchedToday;
                    document.getElementById('referralCode').textContent = user.referral_code;
                    
                    if (user.paypal_email) {
                        document.getElementById('paypalEmail').value = user.paypal_email;
                    }

                    if (user.adsWatchedToday >= 20) {
                        document.getElementById('watchAdBtn').disabled = true;
                        document.getElementById('watchAdBtn').textContent = 'Daily Limit Reached';
                        document.getElementById('watchAdBtn').classList.add('bg-gray-400', 'cursor-not-allowed');
                    }
                }
            } catch (err) {
                console.error('Load profile error:', err);
            }
        }

        async function watchAd() {
            if (adWatching) return;
            adWatching = true;

            const btn = document.getElementById('watchAdBtn');
            btn.disabled = true;
            btn.textContent = 'Loading Ad...';

            // Simulate ad watching (15 seconds)
            let countdown = 15;
            const timer = setInterval(() => {
                btn.textContent = `Watching Ad... ${countdown}s`;
                countdown--;

                if (countdown < 0) {
                    clearInterval(timer);
                    creditAdView();
                }
            }, 1000);
        }

        async function creditAdView() {
            try {
                const response = await fetch(`${API_URL}/ads/credit`, {
                    method: 'POST',
                    headers: {
                        'Authorization': `Bearer ${token}`,
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify({ adType: 'video' })
                });

                const data = await response.json();

                if (data.success) {
                    alert(`âœ… Earned $${data.earned.toFixed(2)}! New balance: $${data.balance.toFixed(2)}`);
                    loadProfile();
                } else {
                    alert(data.error || 'Failed to credit ad view');
                }
            } catch (err) {
                alert('Error: ' + err.message);
            } finally {
                adWatching = false;
                const btn = document.getElementById('watchAdBtn');
                btn.disabled = false;
                btn.textContent = 'Watch Ad Now';
            }
        }

        async function requestWithdrawal() {
            const paypalEmail = document.getElementById('paypalEmail').value;
            
            if (!paypalEmail) {
                alert('Please enter your PayPal email');
                return;
            }

            if (!confirm('Request withdrawal? This will be processed within 24-48 hours.')) {
                return;
            }

            try {
                const response = await fetch(`${API_URL}/withdraw/request`, {
                    method: 'POST',
                    headers: {
                        'Authorization': `Bearer ${token}`,
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify({ method: 'paypal', paypalEmail })
                });

                const data = await response.json();

                if (data.success) {
                    alert(data.message);
                    loadProfile();
                } else {
                    alert(data.error || 'Withdrawal failed');
                }
            } catch (err) {
                alert('Error: ' + err.message);
            }
        }

        function logout() {
            localStorage.clear();
            window.location.href = '/';
        }

        loadProfile();
        setInterval(loadProfile, 30000); // Refresh every 30 seconds
    </script>
